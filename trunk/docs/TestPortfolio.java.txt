			-0.888,  0.704, -0.651, -0.656,  1.000, -0.170,  0.922,  0.353, -0.380,  0.642,

			-0.782,  0.827, -0.886, -0.368,  0.922, -0.045,  1.000,  0.534, -0.448,  0.841,
			 0.057,  0.892, -0.631,  0.111,  0.353, -0.107,  0.534,  1.000, -0.800,  0.384,
			 0.006, -0.685,  0.440,  0.123, -0.380,  0.639, -0.448, -0.800,  1.000, -0.386,

		for (int i = 0; i < numeroProjetos; i++)
